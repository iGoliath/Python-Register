CREATE TABLE Inventory(item_id INTEGER PRIMARY KEY AUTOINCREMENT, Name TEXT NOT NULL, Price REAL NOT NULL, Taxable BOOLEAN, Barcode TEXT, Quantity INTEGER, Category TEXT)
CREATE TABLE sales(sale_id INTEGER PRIMARY KEY, "Non-Tax" REAL,"Pre-Tax" REAL, Tax REAL, Total REAL, "# Items Sold" INTEGER, Date TEXT, Time TEXT, "Cash Used" REAL, "CC Used" REAL, Voided BOOLEAN)
CREATE TABLE saleitems(sale_id INTEGER, Name TEXT, Price REAL, Tax BOOLEAN, Barcode TEXT, Quantity INTEGER)
CREATE TABLE no_sale(date TEXT UNIQUE, times_pressed INT)
CREATE TABLE seasonals(seasonal_id INTEGER PRIMARY KEY AUTOINCREMENT, seasonal_lname TEXT, seasonal_fname TEXT, seasonal_siteno TEXT, seasonal_balance REAL)
CREATE TABLE seasonal_sales(seasonal_sale_id INTEGER PRIMARY KEY AUTOINCREMENT, seasonal_id INTEGER, sale_id INTEGER, FOREIGN KEY(seasonal_id) REFERENCES seasonals(seasonal_id), FOREIGN KEY(sale_id) REFERENCES sales(sale_id))
